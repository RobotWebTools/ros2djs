  5 /**
  6  * A Viewer can be used to render an interactive 2D scene to a HTML5 canvas.
  7  *
  8  * @constructor
  9  * @param options - object with following keys:
 10  *  * divID - the ID of the div to place the viewer in
 11  *  * width - the initial width, in pixels, of the canvas
 12  *  * height - the initial height, in pixels, of the canvas
 13  *  * background - the color to render the background, like #efefef
 14  */
 15 ROS2D.Viewer = function(options) {
 16   var that = this;
 17   var options = options || {};
 18   this.divID = options.divID;
 19   this.width = options.width;
 20   this.height = options.height;
 21   this.background = options.background || '#111111';
 22 
 23   // create the canvas to render to
 24   var canvas = document.createElement('canvas');
 25   canvas.width = this.width;
 26   canvas.height = this.height;
 27   canvas.style.background = this.background;
 28   document.getElementById(this.divID).appendChild(canvas);
 29   // create the easel to use
 30   this.scene = new createjs.Stage(canvas);
 31   
 32   // default zoom factor
 33   this.scene.scaleX = 20;
 34   this.scene.scaleY = 20;
 35   
 36   // change Y axis center
 37   this.scene.y = this.height;
 38 
 39   // add the renderer to the page
 40   document.getElementById(this.divID).appendChild(canvas);
 41   
 42   // update at 30fps
 43   createjs.Ticker.setFPS(30);
 44   createjs.Ticker.addListener(function() {
 45     that.scene.update();
 46   });
 47 };
 48 
 49 /**
 50  * Add the given createjs ojbect to the global scene in the viewer.
 51  * 
 52  * @param object - the object to add
 53  */
 54 ROS2D.Viewer.prototype.addObject = function(object) {
 55   this.scene.addChild(object);
 56 };
