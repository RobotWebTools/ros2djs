  9 // convert the given global Stage coordinates to ROS coordinates
 10 createjs.Stage.prototype.globalToRos = function(x, y) {
 11   var rosX = x / this.scaleX;
 12   // change Y direction
 13   var rosY = (this.y - y) / this.scaleY;
 14   return {
 15     x : rosX,
 16     y : rosY
 17   };
 18 };
 19 
 20 // convert a ROS quaternion to theta in degrees
 21 createjs.Stage.prototype.rosQuaternionToGlobalTheta = function(orientation) {
 22   // convert to radians
 23   var q0 = orientation.w;
 24   var q1 = orientation.x;
 25   var q2 = orientation.y;
 26   var q3 = orientation.z;
 27   var theta = Math.atan2(2 * (q0 * q3 + q1 * q2), 1 - 2 * (Math.pow(q2, 2) + Math.pow(q3, 2)));
 28 
 29   // convert to degrees
 30   var deg = theta * (180.0 / Math.PI);
 31   if (deg >= 0 && deg <= 180) {
 32     deg += 270;
 33   } else {
 34     deg -= 90;
 35   }
 36 
 37   return -deg;
 38 };
