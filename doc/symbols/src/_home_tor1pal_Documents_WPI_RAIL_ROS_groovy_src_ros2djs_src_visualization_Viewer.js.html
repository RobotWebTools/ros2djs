  5 /**
  6  * A Viewer can be used to render an interactive 2D scene to a HTML5 canvas.
  7  *
  8  * @constructor
  9  * @param options - object with following keys:
 10  *  * divID - the ID of the div to place the viewer in
 11  *  * width - the initial width, in pixels, of the canvas
 12  *  * height - the initial height, in pixels, of the canvas
 13  *  * background (optional) - the color to render the background, like #efefef
 14  *  * resolution (optional) - the pixels per meter resolution
 15  */
 16 ROS2D.Viewer = function(options) {
 17   var that = this;
 18   var options = options || {};
 19   this.divID = options.divID;
 20   this.width = options.width;
 21   this.height = options.height;
 22   this.resolution = options.resolution || 0.05;
 23   this.background = options.background || '#111111';
 24 
 25   // create the canvas to render to
 26   var canvas = document.createElement('canvas');
 27   canvas.width = this.width;
 28   canvas.height = this.height;
 29   canvas.style.background = this.background;
 30   document.getElementById(this.divID).appendChild(canvas);
 31   // create the easel to use
 32   this.scene = new createjs.Stage(canvas);
 33 
 34   // change Y axis center
 35   this.scene.y = this.height;
 36 
 37   // add the renderer to the page
 38   document.getElementById(this.divID).appendChild(canvas);
 39 
 40   // update at 30fps
 41   createjs.Ticker.setFPS(30);
 42   createjs.Ticker.addListener(function() {
 43     that.scene.update();
 44   });
 45 };
 46 
 47 /**
 48  * Add the given createjs object to the global scene in the viewer.
 49  * 
 50  * @param object - the object to add
 51  */
 52 ROS2D.Viewer.prototype.addObject = function(object) {
 53   this.scene.addChild(object);
 54 };
 55 
 56 /**
 57  * Scale the scene to fit the given width and height into the current canvas.
 58  * 
 59  * @param width - the width to scale to in meters
 60  * @param height - the height to scale to in meters
 61  */
 62 ROS2D.Viewer.prototype.scaleToDimensions = function(width, height) {
 63   this.scene.scaleX = this.width / width;
 64   this.scene.scaleY = this.height / height;
 65 };
