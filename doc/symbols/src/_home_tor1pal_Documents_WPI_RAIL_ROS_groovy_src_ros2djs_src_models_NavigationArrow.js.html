  5 /**
  6  * A navigation arrow is a directed triangle that can be used to display orientation.
  7  * 
  8  * @constructor
  9  * @param options - object with following keys:
 10  *   * size (optional) - the size of the marker
 11  *   * strokeSize (optional) - the size of the outline
 12  *   * strokeColor (optional) - the createjs color for the stroke
 13  *   * fillColor (optional) - the createjs color for the fill
 14  */
 15 ROS2D.NavigationArrow = function(options) {
 16   var options = options || {};
 17   var size = options.size || 10;
 18   var strokeSize = options.strokeSize || 3;
 19   var strokeColor = options.strokeColor || createjs.Graphics.getRGB(0, 0, 0);
 20   var fillColor = options.fillColor || createjs.Graphics.getRGB(255, 0, 0);
 21 
 22   // draw the arrow
 23   var graphics = new createjs.Graphics();
 24   // line width
 25   graphics.setStrokeStyle(strokeSize);
 26   graphics.moveTo(-size / 2.0, size / 2.0);
 27   graphics.beginStroke(strokeColor);
 28   graphics.beginFill(fillColor);
 29   graphics.lineTo(0, -size);
 30   graphics.lineTo(size / 2.0, size / 2.0);
 31   graphics.lineTo(-size / 2.0, size / 2.0);
 32   graphics.closePath();
 33   graphics.endFill();
 34   graphics.endStroke();
 35 
 36   // create the shape
 37   createjs.Shape.call(this, graphics);
 38 };
 39 ROS2D.NavigationArrow.prototype.__proto__ = createjs.Shape.prototype;
